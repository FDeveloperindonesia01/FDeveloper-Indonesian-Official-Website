// ═══ CONFIG ══════════════════════════════════════ const SERVER_URL = 'http://localhost:3000'; // ═══ STATE ══════════════════════════════════════ let socket = null; let sessions = []; let activeSession = null; let authenticated = false; // ═══ LOGIN ══════════════════════════════════════ function doLogin(){ const pass = document.getElementById('passInput').value; if (!pass) return; if (!socket || !socket.connected) { initSocket(pass); } else { socket.emit('admin_join', { password: pass }); } } function doLogout(){ authenticated = false; document.getElementById('loginScreen').style.display = 'flex'; document.getElementById('app').classList.remove('show'); document.getElementById('passInput').value = ''; } // ═══ SOCKET ══════════════════════════════════════ function initSocket(pass){ try { socket = io(SERVER_URL); socket.on('connect', () => { console.log('Connected to server'); document.getElementById('statConn').textContent = 'Terhubung'; document.getElementById('statConn').style.color = 'var(--green)'; document.getElementById('statSub').textContent = 'Socket.IO aktif'; if (pass) socket.emit('admin_join', { password: pass }); }); socket.on('auth_ok', () => { authenticated = true; document.getElementById('loginScreen').style.display = 'none'; document.getElementById('app').classList.add('show'); document.getElementById('loginErr').textContent = ''; }); socket.on('auth_error', (msg) => { document.getElementById('loginErr').textContent = msg; }); socket.on('sessions_update', (list) => { sessions = list; renderSidebar(); updateStats(); // Refresh chat aktif if (activeSession) { const s = sessions.find(s => s.sessionId === activeSession); if (s) renderMessages(s.messages); } }); socket.on('new_message', ({ sessionId, msg }) => { if (sessionId === activeSession) { appendMessage(msg); } }); socket.on('disconnect', () => { document.getElementById('statConn').textContent = 'Terputus'; document.getElementById('statConn').style.color = 'var(--rose)'; document.getElementById('statSub').textContent = 'Mencoba reconnect...'; }); } catch(e) { document.getElementById('loginErr').textContent = 'Tidak bisa terhubung ke server.'; document.getElementById('statConn').textContent = 'Gagal'; document.getElementById('statConn').style.color = 'var(--rose)'; } } // ═══ SIDEBAR ══════════════════════════════════════ function renderSidebar(){ const list = document.getElementById('sessionList'); if (!sessions.length) { list.innerHTML = `
👥

Belum ada pengguna yang menghubungi

`; return; } const onlineCount = sessions.filter(s => s.online).length; document.getElementById('onlineCount').textContent = onlineCount + ' online'; list.innerHTML = sessions.map(s => { const initials = s.sessionId.slice(-2).toUpperCase(); const preview = s.lastMsg ? escHtml(s.lastMsg.text.slice(0, 30)) + (s.lastMsg.text.length > 30 ? '…' : '') : 'Belum ada pesan'; const isActive = s.sessionId === activeSession; const pip = s.online ? '
' : '
'; const unread = s.unread > 0 ? `
${s.unread}
` : ''; return `
${initials}${pip}
${s.sessionId} ${unread}
${preview}
`; }).join(''); } // ═══ PILIH SESI ═══════════════════════════════════ function selectSession(sessionId){ activeSession = sessionId; const s = sessions.find(s => s.sessionId === sessionId); if (!s) return; // Update sidebar active state document.querySelectorAll('.session-item').forEach(el => el.classList.remove('active')); document.querySelectorAll('.session-item').forEach(el => { if (el.onclick.toString().includes(sessionId)) el.classList.add('active'); }); renderSidebar(); // Update chat topbar const topbar = document.getElementById('chatTopbar'); topbar.style.display = 'flex'; document.getElementById('cuName').textContent = sessionId; document.getElementById('cuAvatar').textContent = sessionId.slice(-2).toUpperCase(); const statusEl = document.getElementById('cuStatus'); if (s.online) { statusEl.textContent = 'Online'; statusEl.className = 'cu-status'; } else { statusEl.textContent = 'Offline'; statusEl.className = 'cu-status off'; } // Show messages document.getElementById('noChatPlaceholder').style.display = 'none'; document.getElementById('chatMsgs').style.display = 'flex'; renderMessages(s.messages); // Enable input const inputArea = document.getElementById('chatInputArea'); inputArea.classList.remove('disabled'); document.getElementById('sendBtn').disabled = false; document.getElementById('replyInput').focus(); } function renderMessages(msgs){ const container = document.getElementById('chatMsgs'); container.innerHTML = ''; if (!msgs.length) { container.innerHTML = `
Belum ada pesan di percakapan ini
`; return; } msgs.forEach(m => appendMessage(m, false)); container.scrollTop = container.scrollHeight; } function appendMessage(msg, scroll = true){ const container = document.getElementById('chatMsgs'); // Hapus placeholder kosong kalau ada const empty = container.querySelector('[style*="flex:1"]'); if (empty) empty.remove(); const isAdmin = msg.from === 'admin'; const time = new Date(msg.time).toLocaleTimeString('id-ID', { hour:'2-digit', minute:'2-digit' }); const who = isAdmin ? 'Admin' : 'Pengguna'; const div = document.createElement('div'); div.className = 'msg-row ' + (isAdmin ? 'from-admin' : 'from-user'); div.innerHTML = `
${escHtml(msg.text)}
${who} · ${time}
`; container.appendChild(div); if (scroll) container.scrollTop = container.scrollHeight; } // ═══ KIRIM BALASAN ════════════════════════════════ function sendReply(){ if (!activeSession || !socket) return; const inp = document.getElementById('replyInput'); const txt = inp.value.trim(); if (!txt) return; socket.emit('admin_reply', { sessionId: activeSession, text: txt }); inp.value = ''; autoResize(inp); } function handleReplyKey(e){ if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); sendReply(); } } // Kirim typing ke user let typingTimeout; document.addEventListener('input', (e) => { if (e.target.id === 'replyInput' && activeSession && socket) { socket.emit('admin_typing', { sessionId: activeSession }); } }); // ═══ STATS ════════════════════════════════════════ function updateStats(){ document.getElementById('statTotal').textContent = sessions.length; document.getElementById('statOnline').textContent = sessions.filter(s => s.online).length; document.getElementById('statUnread').textContent = sessions.reduce((acc,s) => acc + (s.unread||0), 0); } // ═══ HELPERS ══════════════════════════════════════ function autoResize(el){ el.style.height = 'auto'; el.style.height = Math.min(el.scrollHeight, 110) + 'px'; } function escHtml(s){ return (s||'').replace(/&/g,'&').replace(//g,'>').replace(/\n/g,'
'); } // Auto connect on load (login akan handle auth) window.addEventListener('load', () => { initSocket(null); });
